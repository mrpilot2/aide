  main.cpp
)

  PRIVATE
    applicationtest.cpp
)

  INTERFACE
    catch
)

    $<$<CXX_COMPILER_ID:Clang>:CATCH_CONFIG_NO_CPP17_UNCAUGHT_EXCEPTIONS>

  PRIVATE
    catch
    aide
)

  WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
)
